4. Code Example


from sklearn.metrics.pairwise import cosine_similarity
import numpy as np
# Exemple de vecteurs de films (genres : Action, SF, Drame)
films = {
    "Inception": [1, 1, 0],
    "Interstellar": [0, 1, 1],
    "Titanic": [0, 0, 1]
}
# Calcul de la matrice de similarit√©
vecteurs = np.array(list(films.values()))
similarite = cosine_similarity(vecteurs)
# Recommandations pour "Inception"
film_index = 0  # Index de "Inception"
recommandations = sorted(
    [(list(films.keys())[i], similarite[film_index][i])
     for i in range(len(films)) if i != film_index],
    key=lambda x: x[1],
    reverse=True
)
print("Recommandations pour Inception :", recommandations)
